import "fmt"

func main() {

    SetApiKey("__YOUR_API_KEY__")

subscriptionInput := &lago.SubscriptionInput{
    ExternalCustomerID: "5eb02857-a71e-4ea2-bcf9-57d8885990ba",
    PlanCode:           "startup_plan",
    SubscriptionAt:     "2022-08-08T00:00:00Z",

    panic(err)
}

}
