```go Go
import "fmt"
import "github.com/getlago/lago-go-client"

func main() {
  lagoClient := lago.New().SetApiKey("__YOUR_API_KEY__")

  taxListInput := &lago.TaxListInput{
    PerPage: 1,
    Page: 1,
  }

  taxResult, err := lagoClient.Tax().GetList(taxListInput)
  if err != nil {
    panic(err)
  }

  fmt.Println(taxResult)
}
```
