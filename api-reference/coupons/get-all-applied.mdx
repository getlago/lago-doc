import "fmt"
import "github.com/getlago/lago-go-client"

func main() {
lagoClient := lago.New().
    SetApiKey("__YOUR_API_KEY__")

    PerPage: 1,
    Page: 1,
}

couponResult, err := lagoClient.AppliedCoupon().GetList(couponListInput)
if err != nil {
    // Error is *lago.Error
    panic(err)
}

// couponResult is *lago.CouponResult
fmt.Println(couponResult)
}
