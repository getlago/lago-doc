```go Go
import "fmt"
import "github.com/getlago/lago-go-client"

func main() {
lagoClient := lago.New().
    SetApiKey("__YOUR_API_KEY__")

invoice, err := lagoClient.Invoice().Get("lago_id")
if err != nil {
    // Error is *lago.Error
    panic(err)
}

// invoice is *lago.Invoice
fmt.Println(invoice)
}
```
