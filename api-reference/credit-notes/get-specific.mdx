```go Go
import "fmt"
import "github.com/getlago/lago-go-client"

func main() {
lagoClient := lago.New().SetApiKey("__YOUR_API_KEY__")

creditNote, err := lagoClient.CreditNotes().Get("__CREDIT_NOTE_ID__")
if err != nil {
    // Error is *lago.Error
    panic(err)
}

// credit note is *lago.CreditNote
fmt.Println(creditNote)
}
```
